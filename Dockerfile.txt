# ---- 1) Base image: slim Python 3.11 ----
FROM python:3.11-slim

# ---- 2) System packages (HTTPS certs, then clean up) ----
RUN apt-get update && apt-get install -y --no-install-recommends \
    ca-certificates \
 && rm -rf /var/lib/apt/lists/*

# ---- 3) App working dir ----
WORKDIR /app

# ---- 4) Install Python deps ----
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# ---- 5) Copy your app code ----
COPY . .

# ---- 6) Expose port (Render sets $PORT; default 8000 locally) ----
ENV PORT=8000

# ---- 7) Start FastAPI with uvicorn ----
# - host 0.0.0.0 lets it accept traffic from outside the container
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
